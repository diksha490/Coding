class Solution {
    public String removeDuplicates(String S) {
        
        Stack<Character> stck=new Stack<Character>();
        StringBuilder result=new StringBuilder();
        
        for(int i=0;i<S.length();i++)
        {
            char ch=S.charAt(i);
            if(stck.isEmpty())
            {
                stck.push(ch);
                continue;
            }
            if(stck.peek()==ch)
                stck.pop();
            else
                stck.push(ch);
        }
        while(!stck.isEmpty())
        {
            result.append(stck.pop());
           // return result.reverse().toString();
        }
    return result.reverse().toString();
        
    }
}
